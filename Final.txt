include macros2.asm
include number.asm

.MODEL LARGE
.STACK 200h
.386
.387
.DATA

	MAXTEXTSIZE equ 32
	@var1 	DD 0
	@var2 	DD 0
	@var3 	DD 0
	@varString 	DB MAXTEXTSIZE dup (?),'$'
	@_Inicio_del_programa 	DB "Inicio del programa", MAXTEXTSIZE ,10,'$'
	@_Ingrese_un_nro 	DB "Ingrese un nro", MAXTEXTSIZE ,10,'$'
	@_Ingrese_otro_nro 	DB "Ingrese otro nro", MAXTEXTSIZE ,10,'$'
	@_Resultado_de_la_suma 	DB "Resultado de la suma", MAXTEXTSIZE ,10,'$'
	@_El_nro_x2_es 	DB "El nro x2 es", MAXTEXTSIZE ,10,'$'
	@_2_0 	DD 2.0
	@_Ingrese_una_cadena 	DB "Ingrese una cadena", MAXTEXTSIZE ,10,'$'
	@_Se_ha_leido 	DB "Se ha leido", MAXTEXTSIZE ,10,'$'
	@Aux1 	DD 0
	@Aux2 	DD 0

.CODE
.startup
	mov AX,@DATA
	mov DS,AX

	FINIT

	displayString 	@_Inicio_del_programa
	displayString 	@_Ingrese_un_nro
	getFloat 	@var2
	displayString 	@_Ingrese_otro_nro
	getFloat 	@var3
	fld 	@var2
	fld 	@var3
	fadd
	fstp 	@Aux1
	fld 	@Aux1
	fstp 	@var1
	displayString 	@_Resultado_de_la_suma
	displayFloat 	@var1,3
	newLine 1
	displayString 	@_Ingrese_un_nro
	getFloat 	@var1
	displayString 	@_El_nro_x2_es
	fld 	@var1
	fld 	@_2_0
	fmul
	fstp 	@Aux2
	fld 	@Aux2
	fstp 	@var1
	displayFloat 	@var1,3
	newLine 1
	displayString 	@_Ingrese_una_cadena
	getString 	@varString
	displayString 	@_Se_ha_leido
	displayString 	@varString
	mov ah, 4ch
	int 21h
end
